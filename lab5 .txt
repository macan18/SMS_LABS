CREATE FUNCTION ChooseCategory (@CategoryNAme NVARCHAR(40))
RETURNS TABLE
AS
RETURN
(
    SELECT P.ProductID, P.ProductName, P.Price FROM Products P
    INNER JOIN Categories C ON P.CategoryID = C.CategoryID
    WHERE C.CategoryName = @CategoryName
);

GO

CREATE FUNCTION AvgPrice (@CategoryID INT)
RETURNS INT
AS
BEGIN
    RETURN (
        SELECT ISNULL(AVG(Price), 0) 
        FROM Products 
        WHERE CategoryID = @CategoryID
    );
END;
GO

SELECT CategoryName, dbo.AvgPrice(CategoryID) AS AveragePrice FROM Categories;
